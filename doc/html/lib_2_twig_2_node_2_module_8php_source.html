00287     protected function compileIsTraitable(Twig_Compiler $compiler)
00288     {
00289         // A template can be used as a trait if:
00290         //   * it has no parent
00291         //   * it has no macros
00292         //   * it has no body
00293         //
00294         // Put another way, a template can be used as a trait if it
00295         // only contains blocks and use statements.
00296         $traitable = null === $this->getNode('parent') && 0 === count($this->getNode('macros'));
00297         if ($traitable) {
00298             if ($this->getNode('body') instanceof Twig_Node_Body) {
00299                 $nodes = $this->getNode('body')->getNode(0);
00300             } else {
00301                 $nodes = $this->getNode('body');
00302             }
00303 
00304             if (!count($nodes)) {
00305                 $nodes = new Twig_Node(array($nodes));
00306             }
00307 
00308             foreach ($nodes as $node) {
00309                 if (!count($node)) {
00310                     continue;
00311                 }
00312 
00313                 if ($node instanceof Twig_Node_Text && ctype_space($node->getAttribute('data'))) {
00314                     continue;
00315                 }
00316 
00317                 if ($node instanceof Twig_Node_BlockReference) {
00318                     continue;
00319                 }
00320 
00321                 $traitable = false;
00322                 break;
00323             }
00324         }
00325 
00326         if ($traitable) {
00327             return;
00328         }
00329 
00330         $compiler
00331             ->write("public function isTraitable()\n", "{\n")
00332             ->indent()
00333             ->write(sprintf("return %s;\n", $traitable ? 'true' : 'false'))
00334             ->outdent()
00335             ->write("}\n\n")
