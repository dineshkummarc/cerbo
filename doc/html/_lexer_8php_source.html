00104             // dispatch to the lexing functions depending

00106             switch ($this->state) {
00107                 case self::STATE_DATA:
00108                     $this->lexData();
00109                     break;
00110 
00111                 case self::STATE_BLOCK:
00112                     $this->lexBlock();
00113                     break;
00114 
00115                 case self::STATE_VAR:
00116                     $this->lexVar();
00117                     break;
00118 
00119                 case self::STATE_STRING:
00120                     $this->lexString();
00121                     break;
00122

00143     protected function lexData()
00144     {
00145         // if no matches are left we return the rest of the template as simple text token
00146         if ($this->position == count($this->positions[0]) - 1) {
00147             $this->pushToken(Twig_Token::TEXT_TYPE, substr($this->code, $this->cursor));
00148             $this->cursor = $this->end;
00149 
00150             return;
00151         }
00152 
00153         // Find the first token after the current cursor
00154         $position = $this->positions[0][++$this->position];
00155         while ($position[1] < $this->cursor) {
00156             if ($this->position == count($this->positions[0]) - 1) {
00157                 return;
00158             }
00159             $position = $this->positions[0][++$this->position];
00160         }
00161 
00162         // push the template text first
00163         $text = $textContent = substr($this->code, $this->cursor, $position[1] - $this->cursor);
00164         if (isset($this->positions[2][$this->position][0])) {
00165             $text = rtrim($text);
00166         }
00167         $this->pushToken(Twig_Token::TEXT_TYPE, $text);
00168         $this->moveCursor($textContent.$position[0]);
00169 
00170         switch ($this->positions[1][$this->position][0]) {
00171             case $this->options['tag_comment'][0]:
00172                 $this->lexComment();
00173                 break;
00174 
00175             case $this->options['tag_block'][0]:
00176                 // raw data?
00177                 if (preg_match($this->regexes['lex_block_raw'], $this->code, $match, null, $this->cursor)) {
00178                     $this->moveCursor($match[0]);
00179                     $this->lexRawData();

00181                 } elseif (preg_match($this->regexes['lex_block_line'], $this->code, $match, null, $this->cursor)) {
00182                     $this->moveCursor($match[0]);
00183                     $this->lineno = (int) $match[1];
00184                 } else {
00185                     $this->pushToken(Twig_Token::BLOCK_START_TYPE);
00186                     $this->pushState(self::STATE_BLOCK);
00187                 }
00188                 break;
00189 
00190             case $this->options['tag_variable'][0]:
00191                 $this->pushToken(Twig_Token::VAR_START_TYPE);

00220     {
00221         // whitespace
00222         if (preg_match('/\s+/A', $this->code, $match, null, $this->cursor)) {
00223             $this->moveCursor($match[0]);
00224 
00225             if ($this->cursor >= $this->end) {
00226                 throw new Twig_Error_Syntax(sprintf('Unexpected end of file: Unclosed "%s"', $this->state === self::STATE_BLOCK ? 'block' : 'variable'), $this->lineno, $this->filename);
00227             }
00228         }
00229 
00230         // operators
00231         if (preg_match($this->regexes['operator'], $this->code, $match, null, $this->cursor)) {
00232             $this->pushToken(Twig_Token::OPERATOR_TYPE, $match[0]);
00233             $this->moveCursor($match[0]);
00234         }
00235         // names
00236         elseif (preg_match(self::REGEX_NAME, $this->code, $match, null, $this->cursor)) {
00237             $this->pushToken(Twig_Token::NAME_TYPE, $match[0]);
00238             $this->moveCursor($match[0]);
00239         }
00240         // numbers
00241         elseif (preg_match(self::REGEX_NUMBER, $this->code, $match, null, $this->cursor)) {
00242             $number = (float) $match[0];  // floats
00243             if (ctype_digit($match[0]) && $number <= PHP_INT_MAX) {
00244                 $number = (int) $match[0]; // integers lower than the maximum
00245             }
00246             $this->pushToken(Twig_Token::NUMBER_TYPE, $number);
00247             $this->moveCursor($match[0]);
00248         }
00249         // punctuation
00250         elseif (false !== strpos(self::PUNCTUATION, $this->code[$this->cursor])) {
00251             // opening bracket
00252             if (false !== strpos('([{', $this->code[$this->cursor])) {
00253                 $this->brackets[] = array($this->code[$this->cursor], $this->lineno);
00254             }
00255             // closing bracket
00256             elseif (false !== strpos(')]}', $this->code[$this->cursor])) {
00257                 if (empty($this->brackets)) {
00258                     throw new Twig_Error_Syntax(sprintf('Unexpected "%s"', $this->code[$this->cursor]), $this->lineno, $this->filename);
00259                 }
00260 
00261                 list($expect, $lineno) = array_pop($this->brackets);
00262                 if ($this->code[$this->cursor] != strtr($expect, '([{', ')]}')) {
00263                     throw new Twig_Error_Syntax(sprintf('Unclosed "%s"', $expect), $lineno, $this->filename);
00264                 }
00265             }
00266 
00267             $this->pushToken(Twig_Token::PUNCTUATION_TYPE, $this->code[$this->cursor]);
00268             ++$this->cursor;
00269         }
00270         // strings
00271         elseif (preg_match(self::REGEX_STRING, $this->code, $match, null, $this->cursor)) {
00272             $this->pushToken(Twig_Token::STRING_TYPE, stripcslashes(substr($match[0], 1, -1)));
00273             $this->moveCursor($match[0]);
00274         }
00275         // opening double quoted string
00276         elseif (preg_match(self::REGEX_DQ_STRING_DELIM, $this->code, $match, null, $this->cursor)) {
00277             $this->brackets[] = array('"', $this->lineno);
00278             $this->pushState(self::STATE_STRING);
00279             $this->moveCursor($match[0]);
00280         }
00281         // unlexable
00282         else {
00283             throw new Twig_Error_Syntax(sprintf('Unexpected character "%s"', $this->code[$this->cursor]), $this->lineno, $this->filename);
00284         }

00371             array_keys($this->env->getUnaryOperators()),
00372             array_keys($this->env->getBinaryOperators())

00374 
00375         $operators = array_combine($operators, array_map('strlen', $operators));

00380             // an operator that ends with a character must be followed by
